Create employees table (pgTable):
- id: integer().primaryKey().generatedAlwaysAsIdentity()
- fullName: varchar('full_name', { length: 255 }).notNull()
- hireDate: date('hire_date', { mode: 'string' }).notNull()
- isActive: boolean('is_active').default(true).notNull()
- isDeel: boolean('is_deel').default(false).notNull()
- email: varchar('email', { length: 255 })
- createdAt: timestamp('created_at', { withTimezone: true }).defaultNow().notNull()

Create vacationRecords table (pgTable 'vacation_records'):
- id: integer().primaryKey().generatedAlwaysAsIdentity()
- employeeId: integer('employee_id').notNull().references(() => employees.id)
- recordType: recordTypeEnum('record_type').notNull()
- startDate: date('start_date', { mode: 'string' })
- endDate: date('end_date', { mode: 'string' })
- daysCount: real('days_count')
- year: integer('year')
- note: text('note')
- createdAt: timestamp('created_at', { withTimezone: true }).defaultNow().notNull()

Export all: recordTypeEnum, employees, vacationRecords.
Also export inferred types: Employee ($inferSelect), NewEmployee ($inferInsert), VacationRecord ($inferSelect), NewVacationRecord ($inferInsert).

Import from drizzle-orm/pg-core: pgTable, pgEnum, integer, varchar, boolean, date, timestamp, real, text.
